Default options

var myBillboard = new $.billboard(
"#myBillboard",
{ 
 ease: "easeInOutExpo",		// animation ease of transitions
 width: 0,
 height: 0,
 speed: 1000,			// duration of transitions in milliseconds
 duration: 5000,		// time between slide changes
 autoplay: true,		// whether slideshow should play automatically
 loop: true,			// whether slideshow should loop (only applies if autoplay is true)
 transition: "left", 		// "fade", "up", "down", "left", "right"
 nav: "list", 			// "controls", "list", "both" or "none"
 style_nav: true,		// applies default styles to nav
 footer: true,			// show/hide footer (contains caption and nav)
 detect_size: false		// attempts to detect slideshow size automatically	
});

Sample billboards

$(".billboard.left").billboard();
$(".billboard.right").billboard({ 
	transition: "right", 
	duration: 2000,
	speed: 500,
	nav: "controls"
});
$(".billboard.up").billboard({ 
	transition: "up", 
	nav: "none"
});
$(".billboard.up").billboard({ 
	ease: "easeInOutElastic",
	speed: 2000,
	transition: "down"
});
$(".billboard.fade").billboard({ 
	transition: "fade",
	footer: false
});
$(".billboard.goto").billboard();
$("nav.goto a").click(function(e) {
	e.preventDefault();
	$.fn.billboard.goto($("nav a").index(this));
});
